// Safe fetch with timeout — never throws, returns null on failure
async function safeFetch(url, ms = 6000) {
  try {
    const ctrl = new AbortController();
    const timer = setTimeout(() => ctrl.abort(), ms);
    const r = await fetch(url, { signal: ctrl.signal });
    clearTimeout(timer);
    return r.ok ? r : null;
  } catch (_) { return null; }
}

function parseBinanceKline(k) {
  return { o:+k[1], h:+k[2], l:+k[3], c:+k[4], v:+k[5], t:+k[0], done:true };
}

// CoinGecko OHLC format: [timestamp, open, high, low, close]
function parseCGOHLC(row) {
  return { o:+row[1], h:+row[2], l:+row[3], c:+row[4], v:500, t:+row[0], done:true };
}

// Scale CoinGecko bars to match real Binance WS price
// CoinGecko PAXG/XAUT may differ from XAUUSDT by a small ratio
function scaleToPrice(bars, realPrice) {
  if (!bars.length || !realPrice) return bars;
  const lastClose = bars[bars.length-1].c;
  if (lastClose <= 0) return bars;
  const ratio = realPrice / lastClose;
  // Only scale if difference > 1% (avoid scaling identical data)
  if (Math.abs(ratio - 1) < 0.01) return bars;
  return bars.map(b => ({
    ...b,
    o: r2(b.o * ratio), h: r2(b.h * ratio),
    l: r2(b.l * ratio), c: r2(b.c * ratio),
  }));
}

// CoinGecko market_chart format: prices=[[ts,price],...], gives close only
function parseCGChart(prices) {
  if (!prices || prices.length < 2) return [];
  const bars = [];
  for (let i = 1; i < prices.length; i++) {
    const prev = +prices[i-1][1];
    const curr = +prices[i][1];
    const hi   = Math.max(prev, curr) + Math.random() * 1.5;
    const lo   = Math.min(prev, curr) - Math.random() * 1.0;
    bars.push({ o: r2(prev), h: r2(hi), l: r2(lo), c: r2(curr), v:500, t:+prices[i][0], done:true });
  }
  return bars;
}

// Realistic synthetic gold data — always available, instant
function makeSynthetic(n = 200) {
  // Seed with a realistic starting price near current gold
  let p = 3280 + Math.random() * 80;
  const bars = [];
  const now  = Date.now();
  const MS15 = 15 * 60 * 1000;
  let trendDir = 1, trendCount = 0, trendLen = 15 + Math.floor(Math.random() * 25);
  for (let i = n - 1; i >= 0; i--) {
    if (++trendCount > trendLen) {
      trendDir = Math.random() > 0.45 ? 1 : -1;
      trendLen = 12 + Math.floor(Math.random() * 28);
      trendCount = 0;
    }
    const volatility = 2.5 + Math.random() * 2;
    const o = p;
    const m = (Math.random() - 0.487) * volatility + trendDir * 0.35;
    const c = Math.max(3050, Math.min(3650, p + m));
    const wickH = Math.random() * volatility * 0.6;
    const wickL = Math.random() * volatility * 0.4;
    bars.push({
      o: r2(o),
      h: r2(Math.max(o, c) + wickH),
      l: r2(Math.min(o, c) - wickL),
      c: r2(c), v: 300 + Math.floor(Math.random() * 1200),
      t: now - i * MS15, done: true
    });
    p = c;
  }
  return bars;
}

async function loadKlines(limit = 200) {
  // ── 1. Binance direct from browser ─────────────────────
  for (const base of ["https://api.binance.com","https://api1.binance.com","https://api2.binance.com"]) {
    const r = await safeFetch(base+"/api/v3/klines?symbol="+SYMBOL+"&interval="+INTERVAL+"&limit="+limit, 5000);
    if (r) {
      try {
        const d = await r.json();
        if (Array.isArray(d) && d.length > 0) return { bars: d.map(parseBinanceKline), source:"binance" };
      } catch (_) {}
    }
  }

  // ── 2. CoinGecko OHLC (pax-gold = PAXG, tracks XAU 1:1) ─
  const CG_COINS = ["pax-gold", "tether-gold"];
  for (const coin of CG_COINS) {
    const r = await safeFetch(
      "https://api.coingecko.com/api/v3/coins/"+coin+"/ohlc?vs_currency=usd&days=7",
      8000
    );
    if (r) {
      try {
        const d = await r.json();
        if (Array.isArray(d) && d.length > 5) return { bars: d.map(parseCGOHLC), source:"coingecko" };
      } catch (_) {}
    }
  }

  // ── 3. CoinGecko market_chart (close price only, reconstruct OHLC) ──
  for (const coin of CG_COINS) {
    const r = await safeFetch(
      "https://api.coingecko.com/api/v3/coins/"+coin+"/market_chart?vs_currency=usd&days=5&interval=hourly",
      8000
    );
    if (r) {
      try {
        const d = await r.json();
        const bars = parseCGChart(d?.prices);
        if (bars.length > 10) return { bars, source:"coingecko-chart" };
      } catch (_) {}
    }
  }

  // ── 4. Synthetic — instant, always works ────────────────
  return { bars: makeSynthetic(limit), source:"synthetic" };
}

════════════════════════════════════════════════ */
async function apiCall(method, path, body) {
  const opts = { method, headers: { "Content-Type": "application/json" } };
  if (body) opts.body = JSON.stringify(body);
  const res  = await fetch(path, opts);
  const data = await res.json();
  if (!data.ok) throw new Error(data.error || "API error");
  return data;
}

   MATH

const r2  = n => Math.round(n * 100) / 100;
const r1  = n => Math.round(n * 10)  / 10;
const u$  = n => "$" + (n || 0);
const abs$ = n => (n >= 0 ? "+" : "-") + "$" + Math.abs(n);

function emaArr(src, p) {
  const out = new Array(src.length).fill(null);
  if (src.length < p) return out;
  const k = 2 / (p + 1);
  let v = src.slice(0, p).reduce((a, b) => a + b, 0) / p;
  out[p - 1] = v;
  for (let i = p; i < src.length; i++) { v = src[i] * k + v * (1 - k); out[i] = v; }
  return out;
}

function rsiArr(src, p = 14) {
  const out = new Array(src.length).fill(null);
  for (let i = p; i < src.length; i++) {
    let g = 0, l = 0;
    for (let j = i - p + 1; j <= i; j++) {
      const d = src[j] - src[j-1]; d > 0 ? (g += d) : (l -= d);
    }
    const ag = g/p, al = l/p;
    out[i] = r1(al === 0 ? 100 : 100 - 100/(1+ag/al));
  }
  return out;
}

function calcATR(bars, p = 14) {
  if (bars.length < p + 1) return 3;
  const sl  = bars.slice(-p);
  const trs = sl.map((b, i, a) => {
    const pc = i === 0 ? (bars[bars.length-p-1]?.c || b.o) : a[i-1].c;
    return Math.max(b.h-b.l, Math.abs(b.h-pc), Math.abs(b.l-pc));
  });
  return r2(trs.reduce((a,b)=>a+b,0)/p);
}

function calcPivots(bars, lb = 5) {
  const sup = [], res = [];
  for (let i = lb; i < bars.length - lb; i++) {
    const sl = bars.slice(i-lb, i+lb+1);
    if (sl.every(b => b.l >= bars[i].l)) sup.push(r2(bars[i].l));
    if (sl.every(b => b.h <= bars[i].h)) res.push(r2(bars[i].h));
  }
  return { sup:[...new Set(sup)].slice(-4), res:[...new Set(res)].slice(-4) };
}

/* ════════════════════════════════════════════════
   SIGNAL ENGINE
════════════════════════════════════════════════ */
function getSignal(bars) {
  const N = bars.length;
  const BLANK = { sig:"SCAN", strat:"LOADING", score:0, why:"Collecting data…", e9s:[], e21s:[], e50s:[], rsiArr:[], sup:[], res:[], price:0, e9:0, e21:0, e50:0, rsi:50, atr:3 };
  if (N < 15) return BLANK; // lowered — works with CoinGecko's ~40-50 candles

  const cl   = bars.map(b => b.c);
  const p    = cl[N-1];
  const e9s  = emaArr(cl, 9);
  const e21s = emaArr(cl, 21);
  const e50s = emaArr(cl, 50);
  const rS   = rsiArr(cl, 14);
  const atr  = calcATR(bars, 14);
  const { sup, res } = calcPivots(bars, Math.min(5, Math.floor(N/4)));

  const E9   = e9s[N-1]  || p;
  const E21  = e21s[N-1] || p;
  const E21p = e21s[N-2] || E21;
  const E50  = e50s[N-1] || p;
  const RSI  = rS[N-1]   || 50;
  const prev = cl[N-2]   || p;

  const bullTrend = E9 > E21 && E21 > E50;
  const bearTrend = E9 < E21 && E21 < E50;
  const crossUp   = prev < E21p && p >= E21;
  const crossDown = prev > E21p && p <= E21;
  const atSup = sup.some(s => Math.abs(p-s) < atr*0.9);
  const atRes = res.some(r => Math.abs(p-r) < atr*0.9);

  const base = { e9s, e21s, e50s, rsiArr:rS, e9:r2(E9), e21:r2(E21), e50:r2(E50), rsi:RSI, atr, price:p, sup, res };

  let bS=0, sS=0, bF=[], sF=[];
  if (bullTrend)    { bS+=20; bF.push("EMA9>EMA21>EMA50 aligned"); }
  if (crossUp)      { bS+=25; bF.push("Price crossed above EMA21"); }
  if (RSI > 35 && RSI < 55) { bS+=15; bF.push("RSI bullish zone ("+RSI+")"); }
  if (RSI < 35)     { bS+=22; bF.push("RSI oversold ("+RSI+")"); }
  if (atSup)        { bS+=20; bF.push("At pivot support"); }
  if (bearTrend)    { sS+=20; sF.push("EMA9<EMA21<EMA50 aligned"); }
  if (crossDown)    { sS+=25; sF.push("Price crossed below EMA21"); }
  if (RSI > 45 && RSI < 65) { sS+=15; sF.push("RSI bearish zone ("+RSI+")"); }
  if (RSI > 65)     { sS+=22; sF.push("RSI overbought ("+RSI+")"); }
  if (atRes)        { sS+=20; sF.push("At pivot resistance"); }

  const bP = Math.min(95, bS), sP = Math.min(95, sS);

  if (bP >= 55 && bP > sP) {
    const strat = atSup?"S/R BOUNCE":crossUp?"EMA CROSS":"PULLBACK";
    return { ...base, sig:"BUY", strat, score:bP, factors:bF, why:bF.slice(0,3).join(" · "),
      sl:r2(p-atr*1.1), tp1:r2(p+atr*2.0), tp2:r2(p+atr*3.5) };
  }
  if (sP >= 55 && sP > bP) {
    const strat = atRes?"S/R BOUNCE":crossDown?"EMA CROSS":"PULLBACK";
    return { ...base, sig:"SELL", strat, score:sP, factors:sF, why:sF.slice(0,3).join(" · "),
      sl:r2(p+atr*1.1), tp1:r2(p-atr*2.0), tp2:r2(p-atr*3.5) };
  }

  const near = Math.abs(p-E21) < atr*1.2;
  return { ...base, sig:"SCAN", strat:"WATCHING", score:Math.max(bP,sP),
    why:(bullTrend?"BULL ▲":bearTrend?"BEAR ▼":"NEUTRAL")+" · EMA21 "+r2(E21)+" · RSI "+RSI+" · ATR "+r2(atr),
    sl:null, tp1:null, tp2:null };
}

  if(data.length<2) return null;

  const pts=data.map((v,i)=>X(i).toFixed(1)+","+Y(v).toFixed(1)).join(" ");

      <polyline points={poly} fill="none" stroke={col} strokeWidth="2"/>
      <circle cx={xE(pts.length-1)} cy={yE(pts[pts.length-1])} r="3.5" fill={col}/>
      <text x={PL-4} y={yE(start)+3} textAnchor="end" fontSize="7.5" fill="#334155">{"$"+start}</text>
    </svg>
  );
}

function Gauge({ score }) {
  const col=score>=80?"#22c55e":score>=55?"#f59e0b":"#475569";
  const r=32,cx=42,cy=44;
  const xy=d=>({x:cx+r*Math.cos((d-90)*Math.PI/180),y:cy+r*Math.sin((d-90)*Math.PI/180)});
  const s=xy(-120),bg=xy(120),fg=xy(-120+(Math.min(score,100)/100)*240);
  const la=(score/100)*240>180?1:0;
  return(
    <svg viewBox="0 0 84 68" style={{width:84,height:68,flexShrink:0}}>
      <path d={"M "+s.x.toFixed(1)+" "+s.y.toFixed(1)+" A "+r+" "+r+" 0 1 1 "+bg.x.toFixed(1)+" "+bg.y.toFixed(1)} fill="none" stroke="#0d1b2a" strokeWidth="7" strokeLinecap="round"/>
      {score>0&&<path d={"M "+s.x.toFixed(1)+" "+s.y.toFixed(1)+" A "+r+" "+r+" 0 "+la+" 1 "+fg.x.toFixed(1)+" "+fg.y.toFixed(1)} fill="none" stroke={col} strokeWidth="7" strokeLinecap="round"/>}
      <text x={cx} y={cy-5} textAnchor="middle" fontSize="18" fontWeight="900" fill={col}>{score}</text>
      <text x={cx} y={cy+9} textAnchor="middle" fontSize="7.5" fill="#475569">SCORE</text>
    </svg>
  );
}
